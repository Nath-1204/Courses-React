import React, { useState, useEffect, useRef } from 'react';
import './PomodoroClock.css';

const PomodoroClock = () => {
  // États pour les longueurs de pause et session
  const [breakLength, setBreakLength] = useState(5);
  const [sessionLength, setSessionLength] = useState(25);
  
  // États pour le minuteur
  const [timeLeft, setTimeLeft] = useState(25 * 60); // en secondes
  const [timerLabel, setTimerLabel] = useState('Session');
  const [isRunning, setIsRunning] = useState(false);
  const [isSession, setIsSession] = useState(true);
  
  // Références pour l'intervalle et l'audio
  const timerRef = useRef(null);
  const audioRef = useRef(null);

  // Formater le temps en mm:ss
  const formatTime = (seconds) => {
    const mins = Math.floor(seconds / 60);
    const secs = seconds % 60;
    return `${mins.toString().padStart(2, '0')}:${secs.toString().padStart(2, '0')}`;
  };

  // Gestion des longueurs avec limites
  const handleBreakChange = (increment) => {
    if (!isRunning) {
      const newLength = breakLength + increment;
      if (newLength >= 1 && newLength <= 60) {
        setBreakLength(newLength);
        if (!isSession) {
          setTimeLeft(newLength * 60);
        }
      }
    }
  };

  const handleSessionChange = (increment) => {
    if (!isRunning) {
      const newLength = sessionLength + increment;
      if (newLength >= 1 && newLength <= 60) {
        setSessionLength(newLength);
        if (isSession) {
          setTimeLeft(newLength * 60);
        }
      }
    }
  };

  // Démarrer/arrêter le minuteur
  const toggleTimer = () => {
    setIsRunning(!isRunning);
  };

  // Réinitialiser le minuteur
  const resetTimer = () => {
    // Arrêter le minuteur
    if (timerRef.current) {
      clearInterval(timerRef.current);
      timerRef.current = null;
    }
    
    // Réinitialiser les états
    setIsRunning(false);
    setBreakLength(5);
    setSessionLength(25);
    setTimeLeft(25 * 60);
    setTimerLabel('Session');
    setIsSession(true);
    
    // Arrêter et réinitialiser l'audio
    if (audioRef.current) {
      audioRef.current.pause();
      audioRef.current.currentTime = 0;
    }
  };

  // Gérer le compte à rebours
  useEffect(() => {
    if (isRunning) {
      timerRef.current = setInterval(() => {
        setTimeLeft(prevTime => {
          if (prevTime <= 0) {
            // Temps écoulé, basculer entre session et pause
            audioRef.current.play();
            
            if (isSession) {
              // Session terminée, commencer la pause
              setTimerLabel('Break');
              setIsSession(false);
              return breakLength * 60;
            } else {
              // Pause terminée, commencer la session
              setTimerLabel('Session');
              setIsSession(true);
              return sessionLength * 60;
            }
          }
          return prevTime - 1;
        });
      }, 1000);
    } else if (timerRef.current) {
      clearInterval(timerRef.current);
      timerRef.current = null;
    }

    return () => {
      if (timerRef.current) {
        clearInterval(timerRef.current);
      }
    };
  }, [isRunning, isSession, breakLength, sessionLength]);

  // Mettre à jour le titre de la page avec le temps restant
  useEffect(() => {
    document.title = `${formatTime(timeLeft)} - ${timerLabel} | Pomodoro Clock`;
  }, [timeLeft, timerLabel]);

  return (
    <div className="pomodoro-container">
      <header className="app-header">
        <h1><i className="fas fa-clock"></i> Horloge 25 + 5</h1>
        <p className="subtitle">Technique Pomodoro pour une productivité optimale</p>
      </header>

      <main className="clock-app">
        <div className="controls-container">
          {/* Contrôles pour la pause */}
          <div className="control-group">
            <div id="break-label" className="control-label">
              <i className="fas fa-coffee"></i> Durée de la pause
            </div>
            <div className="control-buttons">
              <button
                id="break-decrement"
                className="control-btn"
                onClick={() => handleBreakChange(-1)}
                disabled={isRunning}
              >
                <i className="fas fa-arrow-down"></i>
              </button>
              <div id="break-length" className="control-value">
                {breakLength}
              </div>
              <button
                id="break-increment"
                className="control-btn"
                onClick={() => handleBreakChange(1)}
                disabled={isRunning}
              >
                <i className="fas fa-arrow-up"></i>
              </button>
            </div>
            <div className="control-info">Minutes</div>
          </div>

          {/* Contrôles pour la session */}
          <div className="control-group">
            <div id="session-label" className="control-label">
              <i className="fas fa-briefcase"></i> Durée de la session
            </div>
            <div className="control-buttons">
              <button
                id="session-decrement"
                className="control-btn"
                onClick={() => handleSessionChange(-1)}
                disabled={isRunning}
              >
                <i className="fas fa-arrow-down"></i>
              </button>
              <div id="session-length" className="control-value">
                {sessionLength}
              </div>
              <button
                id="session-increment"
                className="control-btn"
                onClick={() => handleSessionChange(1)}
                disabled={isRunning}
              >
                <i className="fas fa-arrow-up"></i>
              </button>
            </div>
            <div className="control-info">Minutes</div>
          </div>
        </div>

        {/* Affichage du minuteur */}
        <div className="timer-container">
          <div id="timer-label" className="timer-label">
            {timerLabel}
          </div>
          <div id="time-left" className={`time-display ${timeLeft < 60 ? 'warning' : ''}`}>
            {formatTime(timeLeft)}
          </div>
          
          {/* Contrôles du minuteur */}
          <div className="timer-controls">
            <button
              id="start_stop"
              className="timer-btn start-stop-btn"
              onClick={toggleTimer}
            >
              <i className={`fas fa-${isRunning ? 'pause' : 'play'}`}></i>
              {isRunning ? 'Pause' : 'Démarrer'}
            </button>
            <button
              id="reset"
              className="timer-btn reset-btn"
              onClick={resetTimer}
            >
              <i className="fas fa-redo"></i> Réinitialiser
            </button>
          </div>
        </div>

        {/* Indications */}
        <div className="instructions">
          <h3><i className="fas fa-lightbulb"></i> Comment utiliser :</h3>
          <ul>
            <li>Configurez la durée de session (travail) et de pause</li>
            <li>Cliquez sur Démarrer pour lancer le minuteur</li>
            <li>Travaillez pendant la session, reposez-vous pendant la pause</li>
            <li>Le minuteur alterne automatiquement entre session et pause</li>
          </ul>
        </div>
      </main>

      {/* Élément audio pour la sonnerie */}
      <audio
        id="beep"
        ref={audioRef}
        src="https://assets.mixkit.co/sfx/preview/mixkit-alarm-digital-clock-beep-989.mp3"
        preload="auto"
      />

      <footer className="app-footer">
        <p>Projet freeCodeCamp - Horloge 25 + 5</p>
      </footer>
    </div>
  );
};

export default PomodoroClock;
